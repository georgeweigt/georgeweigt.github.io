"Rutherford scattering 2"

"Verify equation (1)"
f = exp(i p / hbar r - r / a) - exp(-i p / hbar r - r / a)
F = integral(f,r)
I = 0 - eval(F,r,0)
check(I == -1 / (i p / hbar - 1 / a) - 1 / (i p / hbar + 1 / a))
f = -m Z alpha c / hbar hbar / (i p) I
check(f == -2 m Z alpha hbar c / (p^2 + (hbar / a)^2))
"ok"

"Verify equation (2)"
p = sqrt(4 m E (1 - cos(theta)))
check(f == -2 m Z alpha hbar c / (4 m E (1 - cos(theta)) + (hbar/a)^2))
"ok"
