\section*{Time-independent Schrodinger equation}

Consider the Schrodinger equation
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi(x,t)=H\psi(x,t)
\end{equation*}

Let $\psi(x,t)$ be separable as
\begin{equation*}
\psi(x,t)=\varphi(t)\phi(x)
\end{equation*}

Let it be the case that for the time derivative we have
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\varphi(t)=E\varphi(t)
\end{equation*}

where $E$ is a constant. Then
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi(x,t)=E\varphi(t)\phi(x)=E\psi(x,t)
\end{equation*}

Hence by the Schrodinger equation
\begin{equation*}
E\psi(x,t)=H\psi(x,t)
\end{equation*}

If $H$ is independent of $t$ then $\varphi(t)$ cancels and
we have the time-independent Schrodinger equation
\begin{equation*}
E\phi(x)=H\phi(x)
\end{equation*}

In summary, let $\psi(x,t)$ be a wavefunction such that
\begin{enumerate}
\item $\psi(x,t)$ solves the Schrodinger equation
\item $\psi(x,t)$ is separable as $\varphi(t)\phi(t)$
\item $i\hbar\partial_t\varphi(t)=E\varphi(t)$ where $E$ is a constant
\item $H$ is independent of $t$
\end{enumerate}

Then $\phi(x)$ solves the time-independent Schrodinger equation
\begin{equation*}
E\phi(x)=H\phi(x)
\end{equation*}

For example, the Hamiltonian for the harmonic oscillator is
\begin{equation*}
H=-\frac{\hbar^2}{2m}\frac{\partial^2}{\partial x^2}+\frac{1}{2}m\omega^2x^2
\end{equation*}

The wavefunctions are
\begin{equation*}
\psi_n(x,t)=C_n\exp\left(-\frac{iE_nt}{\hbar}\right)
\exp\left(\frac{m\omega x^2}{2\hbar}\right)
\frac{d^n}{dx^n}\exp\left(-\frac{m\omega x^2}{\hbar}\right)
\end{equation*}

where $C_n$ is a normalization constant and
\begin{equation*}
E_n=\hbar\omega\left(n+\tfrac{1}{2}\right)
\end{equation*}

Taking the time derivative of $\psi_n(x,t)$ we find
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi_n(x,t)=E_n\psi_n(x,t)
\end{equation*}

Hence
\begin{equation*}
E_n\psi_n(x,t)=H_n\psi_n(x,t)
\end{equation*}

Since $H$ is independent of $t$ we have
\begin{equation*}
E_n\phi_n(x)=H\phi_n(x)
\end{equation*}

where
\begin{equation*}
\phi_n(x)=C_n
\exp\left(\frac{m\omega x^2}{2\hbar}\right)
\frac{d^n}{dx^n}\exp\left(-\frac{m\omega x^2}{\hbar}\right)
\end{equation*}
