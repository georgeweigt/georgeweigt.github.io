\section*{Time-independent Schrodinger equation}

Consider the Schrodinger equation
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\Psi(x,t)=H\Psi(x,t)
\end{equation*}

Let $\Psi(x,t)$ be separable as
\begin{equation*}
\Psi(x,t)=\sum_k\varphi_k(t)\phi_k(x)
\end{equation*}

Let it be the case that for all $k$
\begin{equation*}
i\hbar\frac{d}{dt}\varphi_k(t)\phi_k(x)=E_k\varphi_k(t)\phi_k(x)
\end{equation*}

where all $E_k$ are constant.
Then by the Schrodinger equation we have
\begin{equation*}
E_k\varphi_k(t)\phi_k(x)=H[\varphi_k(t)\phi_k(x)]
\end{equation*}

If $H$ is independent of $t$ then $\phi_k(t)$ cancels and we have
\begin{equation*}
E_k\phi_k(x)=H\phi_k(x)
\end{equation*}

Hence $\sum_k\phi_k(x)$ solves the time-independent Schrodinger equation
\begin{equation*}
\sum_kE_k\phi_k(x)=H\sum_k\phi_k(x)
\end{equation*}

\iffalse

For example, the Hamiltonian for the harmonic oscillator is
\begin{equation*}
H=-\frac{\hbar^2}{2m}\frac{\partial^2}{\partial x^2}+\frac{1}{2}m\omega^2x^2
\end{equation*}

The wavefunctions are
\begin{equation*}
\psi_n(x,t)=C_n\exp\left(-\frac{iE_nt}{\hbar}\right)
\exp\left(\frac{m\omega x^2}{2\hbar}\right)
\frac{d^n}{dx^n}\exp\left(-\frac{m\omega x^2}{\hbar}\right)
\end{equation*}

where $C_n$ is a normalization constant and
\begin{equation*}
E_n=\hbar\omega\left(n+\tfrac{1}{2}\right)
\end{equation*}

Taking the time derivative of $\psi_n(x,t)$ we find
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi_n(x,t)=E_n\psi_n(x,t)
\end{equation*}

Hence
\begin{equation*}
E_n\psi_n(x,t)=H_n\psi_n(x,t)
\end{equation*}

Since $H$ is independent of $t$ we have
\begin{equation*}
E_n\phi_n(x)=H\phi_n(x)
\end{equation*}

where
\begin{equation*}
\phi_n(x)=C_n
\exp\left(\frac{m\omega x^2}{2\hbar}\right)
\frac{d^n}{dx^n}\exp\left(-\frac{m\omega x^2}{\hbar}\right)
\end{equation*}
