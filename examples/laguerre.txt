"Verify Laguerre polynomials"

-- for integer a (k is a local var)

L(x,n,a,k) = (n + a)! sum(k, 0, n, (-x)^k / ((n - k)! (a + k)! k!))

for(n, 0, 2, for(a, 0, n - 1,
 y = L(x,n,a),
 check(x d(y,x,x) + (a + 1 - x) d(y,x) + n y == 0)
))

-- for real or complex a (y is a local var)

L(x,n,a,y) = eval(y^(-a) exp(y) / n! d(exp(-y) y^(n + a), y, n), y, x)

for(n, 0, 2,
 y = L(x,n,a),
 check(x d(y,x,x) + (a + 1 - x) d(y,x) + n y == 0)
)

"ok"
