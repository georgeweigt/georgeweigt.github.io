"Scattering part 1"

meter = "meter"
kilogram = "kilogram"
second = "second"

joule = kilogram meter^2 / second^2

E = joule
hbar = joule second
m = kilogram
r = meter
c = meter / second

"Verify equation (1)"
check(sqrt(m E) / hbar r == 1)
"ok"

"Verify equation (2)"
check((hbar c / E)^2 == meter^2)
"ok"
