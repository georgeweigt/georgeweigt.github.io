\section*{Harmonic oscillator 1}

Let $H$ be the Hamiltonian for a harmonic oscillator defined by $m$ and $\omega$.
\begin{equation*}
H=-\frac{\hbar^2}{2m}\frac{\partial^2}{\partial x^2}+\frac{1}{2}m\omega^2x^2
\end{equation*}

We seek wavefunction $\psi(x,t)$ that solves the Schrodinger equation
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi(x,t)=H\psi(x,t)
\end{equation*}

The solution for $n=0,1,2,\ldots$ is
\begin{equation*}
\psi_n(x,t)=C_n\exp\left(-\frac{iE_nt}{\hbar}\right)
\exp\left(\frac{m\omega x^2}{2\hbar}\right)
\frac{d^n}{dx^n}\exp\left(-\frac{m\omega x^2}{\hbar}\right)
\end{equation*}

where $C_n$ is the normalization constant
\begin{equation*}
C_n=\frac{(-1)^n}{\sqrt{2^nn!}}
\left(\frac{m\omega}{\pi\hbar}\right)^{1/4}
\left(\frac{\hbar}{m\omega}\right)^{n/2}
\end{equation*}

and $E_n$ is the energy level
\begin{equation*}
E_n=\hbar\omega\left(n+\tfrac{1}{2}\right)
\end{equation*}

Taking the time derivative of $\psi_n(x,t)$ we find
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi_n(x,t)=E_n\psi_n(x,t)
\end{equation*}

Hence by the Schrodinger equation we have
\begin{equation*}
E_n\psi_n(x,t)=H\psi_n(x,t)
\end{equation*}

Factor $\psi_n(x,t)$ as
\begin{equation*}
\psi_n(x,t)=\exp\left(-\frac{iE_nt}{\hbar}\right)\psi_n(x)
\end{equation*}

By substitution
\begin{equation*}
E_n\exp\left(-\frac{iE_nt}{\hbar}\right)\psi_n(x)
=H\left[\exp\left(-\frac{iE_nt}{\hbar}\right)\psi_n(x)\right]
\end{equation*}

Since $H$ is independent of $t$ we can factor the right-hand side as
\begin{equation*}
E_n\exp\left(-\frac{iE_nt}{\hbar}\right)\psi_n(x)=\exp\left(-\frac{iE_nt}{\hbar}\right)H\psi_n(x)
\end{equation*}

The exponentials cancel leaving the time-independent Schrodinger equation
\begin{equation*}
E_n\psi_n(x)=H\psi_n(x)
\end{equation*}
