"Zeeman effect"

E(n,l,j,mj) = V0 / n^2 (1 + alpha^2 / n^2 (n / (j + 1/2) - 3/4)) +
              gJ(l,j) mj muB B

gJ(l,j) = 1 + (j (j + 1) - l (l + 1) + 3/4) / (2 j (j + 1))

check(E(2,1,3/2,3/2) == V0/4 (1 + 1/16 alpha^2) + 2 muB B)
check(E(2,1,3/2,-3/2) == V0/4 (1 + 1/16 alpha^2) - 2 muB B)

check(E(2,1,3/2,1/2) == V0/4 (1 + 1/16 alpha^2) + 2/3 muB B)
check(E(2,1,3/2,-1/2) == V0/4 (1 + 1/16 alpha^2) - 2/3 muB B)

check(E(2,1,1/2,1/2) == V0/4 (1 + 5/16 alpha^2) + 1/3 muB B)
check(E(2,1,1/2,-1/2) == V0/4 (1 + 5/16 alpha^2) - 1/3 muB B)

check(E(2,0,1/2,1/2) == V0/4 (1 + 5/16 alpha^2) + muB B)
check(E(2,0,1/2,-1/2) == V0/4 (1 + 5/16 alpha^2) - muB B)

"ok"

V0 = -13.59828725 "eV"
alpha = 7.2973525643 10^(-3)

V0/4 (1 + 1/16 alpha^2)
V0/4 (1 + 5/16 alpha^2)
