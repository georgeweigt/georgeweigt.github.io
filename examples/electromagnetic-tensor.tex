\section*{Electromagnetic tensor}

This is the standard model for an electromagnetic field.
\begin{align*}
\mathbf A&=\begin{pmatrix}A_x\\A_y\\A_z\end{pmatrix}\qquad
\mathbf B=\nabla\times\mathbf A\qquad
\mathbf E=-\nabla\phi-\frac{\partial\mathbf A}{\partial t}
\\[1ex]
A^\mu&=\begin{pmatrix}\phi\\A_x\\A_y\\A_z\end{pmatrix}\qquad
A_\mu=g_{\mu\nu}A^\nu=\begin{pmatrix}\phi\\-A_x\\-A_y\\-A_z\end{pmatrix}
\end{align*}

This is the electromagnetic tensor.
\begin{equation*}
F_{\mu\nu}
=\partial_\mu A_\nu-\partial_\nu A_\mu
=A_{\nu,\mu}-A_{\mu,\nu}
=\begin{pmatrix}
0 & E_x & E_y & E_z
\\
-E_x & 0 & -B_z & B_y
\\
-E_y & B_z & 0 & -B_x
\\
-E_z & -B_y & B_x & 0
\end{pmatrix}
\tag{1}
\end{equation*}

Verify that
\begin{equation*}
F_{\mu\nu}F^{\mu\nu}=2\bigl(\mathbf B^2-\mathbf E^2\bigl)
\tag{2}
\end{equation*}

Verify that
\begin{equation*}
\det(F_{\mu\nu})=\det(F^{\mu\nu})=(\mathbf B\cdot\mathbf E)^2
\tag{3}
\end{equation*}

This is the vector current.
\begin{equation*}
J^\nu
=\partial_\mu F^{\mu\nu}
=F^{\mu\nu}{}_{,\mu}
\end{equation*}

Verify that
\begin{equation*}
\partial_\mu J^\mu=J^\mu{}_{,\mu}=0
\tag{4}
\end{equation*}

Verify that
\begin{equation*}
\mathbf J=\nabla\times\mathbf B-\frac{\partial\mathbf E}{\partial t}
\tag{5}
\end{equation*}
