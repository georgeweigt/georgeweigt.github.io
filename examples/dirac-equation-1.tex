\section*{Dirac equation 1}

From Fitzpatrick problem 11.11.

\bigskip
Let
\begin{equation*}
\psi^r=\exp\left(-\frac{i}{\hbar}\epsilon_rp_\mu x^\mu\right)w^r(\mathbf p)
\end{equation*}

where $\epsilon=(1,1,-1,-1)$ and
\begin{align*}
w^1(\mathbf p)&=\sqrt{\frac{E+mc^2}{2mc^2}}
\begin{pmatrix}1\\0\\\frac{p_zc}{E+mc^2}\\\frac{(p_x+ip_y)c}{E+mc^2}\end{pmatrix}
&
w^2(\mathbf p)&=\sqrt{\frac{E+mc^2}{2mc^2}}
\begin{pmatrix}0\\1\\\frac{(p_x-ip_y)c}{E+mc^2}\\\frac{-p_zc}{E+mc^2}\end{pmatrix}
\\[1ex]
w^3(\mathbf p)&=\sqrt{\frac{E+mc^2}{2mc^2}}
\begin{pmatrix}\frac{p_zc}{E+mc^2}\\\frac{(p_x+ip_y)c}{E+mc^2}\\1\\0\end{pmatrix}
&
w^4(\mathbf p)&=\sqrt{\frac{E+mc^2}{2mc^2}}
\begin{pmatrix}\frac{(p_x-ip_y)c}{E+mc^2}\\\frac{-p_zc}{E+mc^2}\\0\\1\end{pmatrix}
\end{align*}

Verify that $\psi^r$ are solutions to the Dirac equation
\begin{equation*}
i\hbar\gamma^\mu\partial_\mu\psi=mc\psi
\end{equation*}

and that the solutions are normalized as
\begin{equation*}
|\psi|^2=\frac{E}{mc^2}
\end{equation*}

Note that
\begin{equation*}
\partial_0=\frac{\partial}{\partial(ct)}=\frac{1}{c}\frac{\partial}{\partial t}
\end{equation*}

The gradient $\partial_\mu$ is a covariant vector hence
\begin{equation*}
\gamma^\mu\partial_\mu
=\frac{1}{c}\gamma^0\frac{\partial}{\partial t}
+\gamma^1\frac{\partial}{\partial x}
+\gamma^2\frac{\partial}{\partial y}
+\gamma^3\frac{\partial}{\partial z}
\end{equation*}
