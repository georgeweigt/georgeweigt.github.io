\section*{Born approximation 1}

Derive the Born approximation
\begin{equation*}
\psi(r,\theta,\phi)=-\frac{m}{2\pi\hbar^2}\frac{\exp(ikr)}{r}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf x}{\hbar}\right)V(\mathbf x)\,d\mathbf x
\end{equation*}

where
\begin{equation*}
k=\frac{\sqrt{2mE}}{\hbar}
\end{equation*}

Vector $\mathbf p$ is momentum transfer such that
\begin{equation*}
\mathbf p=\mathbf p_i-\mathbf p_f
\end{equation*}

and
\begin{equation*}
p^2=|\mathbf p|^2=4mE(1-\cos\theta)
\end{equation*}

\subsubsection*{\S1}

Start with the Schrodinger equation
\begin{equation*}
-\frac{\hbar^2}{2m}\nabla^2\psi(\mathbf x)+V(\mathbf x)\psi(\mathbf x)=E\psi(\mathbf x)
\end{equation*}

Multiply through by $-2m/\hbar^2$ to obtain
\begin{equation*}
\nabla^2\psi(\mathbf x)-\frac{2m}{\hbar^2}V(\mathbf x)\psi(\mathbf x)=-\frac{2mE}{\hbar^2}\psi(\mathbf x)
\end{equation*}

Interchange $E$ and $V$ terms.
\begin{equation*}
\nabla^2\psi(\mathbf x)+\frac{2mE}{\hbar^2}\psi(\mathbf x)
=\frac{2m}{\hbar^2}V(\mathbf x)\psi(\mathbf x)
\end{equation*}

Let
\begin{equation*}
k=\frac{\sqrt{2mE}}{\hbar}
\end{equation*}

and rewrite as
\begin{equation*}
\left(\nabla^2+k^2\right)\psi(\mathbf x)=\frac{2m}{\hbar^2}V(\mathbf x)\psi(\mathbf x)
\end{equation*}

By the sifting property of $\delta(\mathbf x-\mathbf y)$ we can make the right-hand
side an integral over $\mathbf y$.
\begin{equation*}
\left(\nabla^2+k^2\right)\psi(\mathbf x)=\frac{2m}{\hbar^2}
\int\delta(\mathbf x-\mathbf y)V(\mathbf y)\psi(\mathbf y)\,d\mathbf y
\end{equation*}

Let $G(\mathbf x)$ be a Green's function such that
\begin{equation*}
\left(\nabla^2+k^2\right)G(\mathbf x)=\delta(\mathbf x)
\end{equation*}

Then by substitution
\begin{equation*}
\left(\nabla^2+k^2\right)\psi(\mathbf x)=\frac{2m}{\hbar^2}
\int\left(\nabla^2+k^2\right)G(\mathbf x-\mathbf y)V(\mathbf y)\psi(\mathbf y)\,d\mathbf y
\end{equation*}

By linearity the integral can be factored as
\begin{equation*}
\left(\nabla^2+k^2\right)\psi(\mathbf x)=\left(\nabla^2+k^2\right)\frac{2m}{\hbar^2}
\int G(\mathbf x-\mathbf y)V(\mathbf y)\psi(\mathbf y)\,d\mathbf y
\end{equation*}

Hence
\begin{equation*}
\psi(\mathbf x)=\frac{2m}{\hbar^2}
\int G(\mathbf x-\mathbf y)V(\mathbf y)\psi(\mathbf y)\,d\mathbf y
\end{equation*}

The solution for $G(\mathbf x)$ is
\begin{equation*}
G(\mathbf x)=-\frac{\exp(ik|\mathbf x|)}{4\pi|\mathbf x|}
\end{equation*}

Hence
\begin{equation*}
\psi(\mathbf x)=-\frac{m}{2\pi\hbar^2}
\int\frac{\exp(ik|\mathbf x-\mathbf y|)}{|\mathbf x-\mathbf y|}\,V(\mathbf y)\psi(\mathbf y)\,d\mathbf y
\end{equation*}

For scattering experiments we can use the approximation
\begin{equation*}
\frac{\exp(ik|\mathbf x-\mathbf y|)}{|\mathbf x-\mathbf y|}\,V(\mathbf y)
\approx\frac{1}{|\mathbf x|}
\exp\left(ik|\mathbf x|-\frac{ik\mathbf x\cdot\mathbf y}{|\mathbf x|}\right)V(\mathbf y)
\end{equation*}

Hence
\begin{equation*}
\psi(\mathbf x)=-\frac{m}{2\pi\hbar^2}\frac{\exp(ik|\mathbf x|)}{|\mathbf x|}
\int\exp\left(-\frac{ik\mathbf x\cdot\mathbf y}{|\mathbf x|}\right)
V(\mathbf y)\psi(\mathbf y)\,d\mathbf y
\end{equation*}

For a first-order approximation we substitute a plane wave for $\psi(\mathbf y)$.
\begin{equation*}
\psi(\mathbf y)=\exp\left(\frac{ik\mathbf x'\cdot\mathbf y}{|\mathbf x'|}\right)
\end{equation*}

Hence
\begin{equation*}
\psi(\mathbf x)=-\frac{m}{2\pi\hbar^2}\frac{\exp(ik|\mathbf x|)}{|\mathbf x|}
\int\exp\left(\frac{ik\mathbf x'\cdot\mathbf y}{|\mathbf x'|}
-\frac{ik\mathbf x\cdot\mathbf y}{|\mathbf x|}\right)
V(\mathbf y)\,d\mathbf y
\end{equation*}

Let $\mathbf p$ be momentum transfer
\begin{equation*}
\mathbf p=\mathbf p_i-\mathbf p_f
=\hbar k\left(\frac{\mathbf x'}{|\mathbf x'|}-\frac{\mathbf x}{|\mathbf x|}\right)
\end{equation*}

Then by substitution
\begin{equation*}
\psi(\mathbf x)=-\frac{m}{2\pi\hbar^2}\frac{\exp(ik|\mathbf x|)}{|\mathbf x|}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf y}{\hbar}\right)
V(\mathbf y)\,d\mathbf y
\end{equation*}

In polar coordinates
\begin{equation*}
\psi(r,\theta,\phi)=-\frac{m}{2\pi\hbar^2}\frac{\exp(ikr)}{r}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf y}{\hbar}\right)
V(\mathbf y)\,d\mathbf y
\end{equation*}

\subsubsection*{\S2}

We now consider the differential cross section $d\sigma$ such that
\begin{equation*}
d\sigma=|\psi|^2\,r^2\,d\Omega
\end{equation*}

For the Born approximation of $\psi$ we have
\begin{equation*}
|\psi|^2=\left|-\frac{m}{2\pi\hbar^2}\frac{\exp(ikr)}{r}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf x}{\hbar}\right)V(\mathbf x)\,d\mathbf x\right|^2
\end{equation*}

The factor $|\exp(ikr)/r|^2=1/r^2$ cancels with $r^2$ in $d\sigma$ hence
\begin{equation*}
d\sigma=\left|-\frac{m}{2\pi\hbar^2}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf x}{\hbar}\right)V(\mathbf x)\,d\mathbf x\right|^2
\,d\Omega
\end{equation*}

Let $f(\theta,\phi)$ be the scattering amplitude
\begin{equation*}
f(\theta,\phi)=-\frac{m}{2\pi\hbar^2}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf x}{\hbar}\right)V(\mathbf x)\,d\mathbf x
\end{equation*}

so that for the Born approximation
\begin{equation*}
\frac{d\sigma}{d\Omega}=|f(\theta,\phi)|^2
\end{equation*}

For example, for Coulomb potentials we have
\begin{equation*}
f(\theta,\phi)=\frac{Z\alpha\hbar c}{2E(1-\cos\theta)}
\end{equation*}

Hence the Rutherford cross section
\begin{equation*}
\frac{d\sigma}{d\Omega}
=|f(\theta,\phi)|^2
=\left[\frac{Z\alpha\hbar c}{2E(1-\cos\theta)}\right]^2
\end{equation*}
