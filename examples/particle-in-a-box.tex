\section*{Particle in a box}

The ``box'' is a potential $V(x)$ such that
\begin{equation*}
V(x)=\begin{cases}
0 & 0\le x\le a
\\
\infty & \text{otherwise}
\end{cases}
\end{equation*}

We seek $\psi(x,t)$ that solves the Schrodinger equation
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi(x,t)
=-\frac{\hbar^2}{2m}\frac{\partial^2}{\partial x^2}\psi(x,t)+V(x)\psi(x,t)
\end{equation*}

For $V(x)=0$ we have the solution
\begin{equation*}
\psi(x,t)=[A\sin(kx)+B\cos(kx)]\exp(-i\omega t)
\end{equation*}

where
\begin{equation*}
\omega=\frac{k^2\hbar}{2m}
\end{equation*}

By the boundary condition
\begin{equation*}
\psi(0,t)=\psi(a,t)=0
\end{equation*}

we obtain $B=0$ and
\begin{equation*}
k=\frac{n\pi}{a}
\end{equation*}

where $n$ is an integer.

\bigskip
We seek $A$ such that the probability of finding the particle in the box is unity.
\begin{equation*}
\int_0^a\psi^*\psi\,dx=1
\end{equation*}

Hence
\begin{equation*}
A=\sqrt{\frac{2}{a}}
\end{equation*}

Let
\begin{equation*}
E_n=\hbar\omega=\frac{n^2\pi^2\hbar^2}{2ma^2}
\end{equation*}

and write
\begin{equation*}
\psi_n(x,t)=\sqrt{\frac{2}{a}}\sin\left(\frac{n\pi x}{a}\right)
\exp\left(-\frac{iE_nt}{\hbar}\right)
\end{equation*}
