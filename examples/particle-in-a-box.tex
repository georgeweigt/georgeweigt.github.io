\section*{Particle in a box}

The ``box'' is a potential $V(x)$ such that
\begin{equation*}
V(x)=\begin{cases}
0 & 0\le x\le a
\\
\infty & \text{otherwise}
\end{cases}
\end{equation*}

We seek $\psi(x,t)$ that solves the Schrodinger equation
\begin{equation*}
i\hbar\frac{\partial}{\partial t}\psi(x,t)
=-\frac{\hbar^2}{2m}\frac{\partial^2}{\partial x^2}\psi(x,t)+V(x)\psi(x,t)
\end{equation*}

For $V(x)=0$ we have the solution
\begin{equation*}
\psi(x,t)=[A\sin(kx)+B\cos(kx)]\exp(-i\omega t)
\end{equation*}

where
\begin{equation*}
\omega=\frac{k^2\hbar}{2m}
\end{equation*}

By the boundary condition $\psi(0,t)=0$ we have
\begin{equation*}
B=0
\end{equation*}

By the boundary condition $\psi(a,t)=0$ we have for integer $n$
\begin{equation*}
kx=\frac{n\pi x}{a}
\end{equation*}

We seek $A$ that normalizes probability density $|\psi|^2$.
\begin{equation*}
\int_0^a|\psi|^2\,dx=\frac{a}{2}\,|A|^2=1
\end{equation*}

Hence
\begin{equation*}
|A|=\sqrt{\frac{2}{a}}
\end{equation*}

Let $E_n$ be particle energy $\hbar\omega$.
\begin{equation*}
E_n=\hbar\omega=\frac{n^2\pi^2\hbar^2}{2ma^2}
\end{equation*}

Put everything together and write
\begin{equation*}
\psi_n(x,t)=\sqrt{\frac{2}{a}}\sin\biggl(\frac{n\pi x}{a}\biggr)
\exp\left(-\frac{iE_nt}{\hbar}\right)
\end{equation*}

Check dimensions.
\begin{align*}
\omega&=\frac{k^2\hbar}{2m}
=\frac{[\text{m}^{-2}]\,[\text{J}\,\text{s}]}{[\text{kg}]}=[\text{s}^{-1}]
\\[1ex]
E_n&=\frac{n^2\pi^2\hbar^2}{2ma^2}
=\frac{[\text{J}^2\,\text{s}^2]}{[\text{kg}]\,[\text{m}^2]}=[\text{J}]
\end{align*}
