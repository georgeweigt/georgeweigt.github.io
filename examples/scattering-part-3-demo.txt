"Verify equation (1)"
f = i / (-i p / hbar - epsilon) - i / (i p / hbar - epsilon)
check(f == -2 hbar p / (p^2 + epsilon^2 hbar^2))
"ok"

"Verify equation (2)"
f = -2 m / (hbar p) sin(p r / hbar) V r
V = beta / r exp(-epsilon r)
f = expform(f)
f = integral(f,r)
f = 0 - eval(f,r,0)
check(f == -2 m beta / (p^2 + epsilon^2 hbar^2))
"ok"
