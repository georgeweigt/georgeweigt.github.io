"Exercise 1. Verify spin operators"

zp = (1,0)
zm = (0,1)

xp = (zp + zm) / sqrt(2)
xm = (zp - zm) / sqrt(2)

yp = (zp + i zm) / sqrt(2)
ym = (zp - i zm) / sqrt(2)

Sx = hbar / 2 ((0,1),(1,0))
Sy = hbar / 2 ((0,-i),(i,0))
Sz = hbar / 2 ((1,0),(0,-1))

check(Sx == hbar / 2 (outer(xp,conj(xp)) - outer(xm,conj(xm))))
check(Sy == hbar / 2 (outer(yp,conj(yp)) - outer(ym,conj(ym))))
check(Sz == hbar / 2 (outer(zp,conj(zp)) - outer(zm,conj(zm))))

"ok"
