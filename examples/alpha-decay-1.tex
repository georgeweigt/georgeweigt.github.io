\section*{Alpha decay 1}

Let $V(r)$ be an ansatz potential for alpha decay
where $2Z$ is the product of alpha-particle charge and daughter nucleus charge.
\begin{equation*}
V(r)=\begin{cases}
-V_0 & 0<r<r_1
\\[1ex]
{\displaystyle \frac{2Z\alpha\hbar c}{r}} & r>r_1
\end{cases}
\end{equation*}

Let $r_2$ be a radius such that $r_2>r_1$ and
\begin{equation*}
V(r_2)=\frac{2Z\alpha\hbar c}{r_2}=E
\end{equation*}

Hence the region from $r_1$ to $r_2$ is a barrier potential for an alpha particle with energy $E$.

\bigskip
Using the WKB method we have
\begin{equation*}
T\approx e^{-2\gamma}
\end{equation*}

where
\begin{equation*}
\gamma=\frac{\sqrt{2m}}{\hbar}\int_{r_1}^{r_2}\sqrt{V(r)-E}\,dr
\end{equation*}

Factor out $\sqrt E$ to obtain
\begin{align*}
\gamma&=\frac{\sqrt{2mE}}{\hbar}\int_{r_1}^{r_2}\sqrt{\frac{V(r)}{E}-1}\,dr
\\[1ex]
&=\frac{\sqrt{2mE}}{\hbar}\int_{r_1}^{r_2}\sqrt{\frac{r_2}{r}-1}\,dr
\end{align*}

Use the substitution
\begin{equation*}
r=r_2\sin^2u
\qquad
dr=2r_2\sin u\cos u
\end{equation*}

to obtain
\begin{align*}
\gamma&=\frac{\sqrt{2mE}}{\hbar}\,2r_2\int_{u_1}^{u_2}
\sqrt{\frac{1}{\sin^2u}-1}\,\sin u\cos u\,du
\\[1ex]
&=\frac{\sqrt{2mE}}{\hbar}\,2r_2\int_{u_1}^{u_2}\sqrt{1-\sin^2u}\,\cos u\,du
\\[1ex]
&=\frac{\sqrt{2mE}}{\hbar}\,2r_2\int_{u_1}^{u_2}\cos^2u\,du
\end{align*}

where
\begin{equation*}
u_1=\arcsin\sqrt{r_1/r_2}
\qquad
u_2=\frac{\pi}{2}
\end{equation*}

Solve the integral to obtain
\begin{equation*}
\gamma=\frac{\sqrt{2mE}}{\hbar}\,2r_2\left[\frac{u}{2}+\frac{\sin(2u)}{4}\right]_{u_1}^{u_2}
\tag{1}
\end{equation*}

Evaluate the limits.
\begin{equation*}
\gamma=\frac{\sqrt{2mE}}{\hbar}\,2r_2
\left[
\frac{\pi}{4}
-\frac{1}{2}\arcsin\sqrt{r_1/r_2}
-\frac{1}{4}\sin\left(2\arcsin\sqrt{r_1/r_2}\,\right)
\right]
\tag{2}
\end{equation*}

From the identities
\begin{equation*}
\sin(2x)=2\sin x\cos x
\end{equation*}

and
\begin{equation*}
\cos(\arcsin x)=\sqrt{1-x^2}
\end{equation*}

the final term simplifies as
\begin{equation*}
\frac{1}{4}\sin\left(2\arcsin\sqrt{r_1/r_2}\,\right)
=\frac{1}{2}\sqrt{r_1/r_2}\sqrt{1-r_1/r_2}
\end{equation*}

Hence
\begin{equation*}
\gamma=\frac{\sqrt{2mE}}{\hbar}\,2r_2
\left(
\frac{\pi}{4}
-\frac{1}{2}\arcsin\sqrt{r_1/r_2}
-\frac{1}{2}\sqrt{r_1/r_2}\sqrt{1-r_1/r_2}\,
\right)
\end{equation*}

Noting that $r_2\gg r_1$ we introduce the approximations
\begin{align*}
\arcsin\sqrt{r_1/r_2}&\approx\sqrt{r_1/r_2}
\\[1ex]
\sqrt{r_1/r_2}\sqrt{1-r_1/r_2}&\approx\sqrt{r_1/r_2}
\end{align*}

and write
\begin{equation*}
\gamma=\frac{\sqrt{2mE}}{\hbar}\,2r_2
\left(
\frac{\pi}{4}
-\sqrt{r_1/r_2}\,
\right)
\end{equation*}

Distribute the $r_2$.
\begin{equation*}
\gamma=\frac{2\sqrt{2mE}}{\hbar}
\left(
\frac{\pi r_2}{4}
-\sqrt{r_1r_2}\,
\right)
\end{equation*}

Substitute
\begin{equation*}
r_2=\frac{2Z\alpha\hbar c}{E}
\end{equation*}

to obtain
\begin{equation*}
\gamma=C_1\frac{Z}{\sqrt E}-C_2\sqrt{Zr_1}
\tag{3}
\end{equation*}

where
\begin{equation*}
C_1=\sqrt{2m}\,\pi\alpha c
\qquad
C_2=4\sqrt\frac{m\alpha c}{\hbar}
\end{equation*}

For the alpha decay process
${}^{238}_{\phantom{0}92}\mathrm U
\rightarrow{}^{234}_{\phantom{0}90}\mathrm{Th}$ we have
\begin{align*}
A&=238
\\
Z&=90
\\
E&=4.27\,\text{MeV}
\end{align*}

Hence
\begin{align*}
r_1&=A^{1/3}\times1.2\times10^{-15}\,\text{m} & &=7.44\times10^{-15}\,\text{m}
\\
r_2&=\frac{2Z\alpha\hbar c}{E} & &=60.7\times10^{-15}\,\text{m}
\end{align*}

Then for $m=4.0015\,\text{u}$ we obtain
\begin{equation*}
\gamma=47.8
\end{equation*}

and
\begin{equation*}
T=e^{-2\gamma}=3.17\times10^{-42}
\end{equation*}
