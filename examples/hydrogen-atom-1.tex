\section*{Hydrogen atom 1}

The first-order Hamiltonian is
\begin{equation*}
H=-\frac{\hbar^2}{2\mu}\nabla^2-\frac{e^2}{4\pi\varepsilon_0r}
\end{equation*}

where $\mu$ is the reduced electron mass
\begin{equation*}
\mu=\frac{m_e m_p}{m_e+m_p}
\end{equation*}

Symbol $\nabla^2$ is the Laplacian operator
\begin{equation*}
\nabla^2=\frac{1}{r^2}\frac{\partial}{\partial r}
\left(r^2\frac{\partial}{\partial r}\right)
+
\frac{1}{r^2\sin\theta}\frac{\partial}{\partial\theta}
\left(\sin\theta\frac{\partial}{\partial\theta}\right)
+
\frac{1}{r^2\sin^2\theta}\frac{\partial^2}{\partial\phi^2}
\end{equation*}

We seek eigenfunctions $\psi(\mathbf r)$ that solve the eigenvalue equation
\begin{equation*}
H\psi(\mathbf r)=E\psi(\mathbf r)
\end{equation*}

The eigenfunctions are
\begin{equation*}
\psi_{nlm}(\mathbf r)=R_{nl}(r)Y_{lm}(\theta,\phi)
\end{equation*}

where
\begin{align*}
n&=1,2,3,\ldots
\\
l&=0,1,2,\ldots,n-1
\\
m&=0,\pm1,\pm2,\ldots,\pm l
\end{align*}

and the eigenvalues are
\begin{equation*}
E_n=-\frac{\mu}{2\hbar^2}\left(\frac{e^2}{4\pi\varepsilon_0}\right)^2\frac{1}{n^2}
\end{equation*}

For $R_{nl}(r)$ we have
\begin{equation*}
R_{nl}(r)=
\frac{2}{n^2}
\sqrt{\frac{(n-l-1)!}{(n+l)!}}
\left(\frac{2r}{na_0}\right)^l
L_{n-l-1}^{2l+1}\left(\frac{2r}{na_0}\right)
\exp\left(-\frac{r}{na_0}\right)
a_0^{-3/2}
\end{equation*}

where $a_0$ is the Bohr radius
\begin{equation*}
a_0=\frac{4\pi\varepsilon_0\hbar^2}{\mu e^2}
\end{equation*}

and $L_n^m$ is the Laguerre polynomial
\begin{equation*}
L_n^m(x)=(n+m)!\sum_{k=0}^n
\frac{(-x)^k}{(n-k)!(m+k)!k!}
\end{equation*}

Symbol $Y_{lm}$ is the spherical harmonic
\begin{equation*}
Y_{lm}(\theta,\phi)=(-1)^m
\sqrt{\frac{(2l+1)}{4\pi}
\frac{(l-m)!}{(l+m)!}}\,
P_l^m(\cos\theta)\exp(im\phi)
\end{equation*}

where Legendre polynomial $P_l^m(\cos\theta)$ is formed as (see arxiv.org/abs/1805.12125)
\begin{equation*}
P_l^m(\cos\theta)=\begin{cases}
\displaystyle
\left(\frac{\sin\theta}{2}\right)^m\,\sum_{k=0}^{l-m}
(-1)^k\frac{(l+m+k)!}{(l-m-k)!(m+k)!k!}
\left(\frac{1-\cos\theta}{2}\right)^k & m\ge0
\\[4ex]
\displaystyle
(-1)^m\frac{(l+m)!}{(l-m)!}P_l^{|m|}(\cos\theta) & m<0
\end{cases}
\end{equation*}
