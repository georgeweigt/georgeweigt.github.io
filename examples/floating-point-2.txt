"Verify 0x3dcccccd is the best approximation for 0.1"

A = 100000001490116119384765625 / 10^27 -- 0x3dcccccd from strtof
B = 099999994039535522460937500 / 10^27 -- 0x3dcccccc

C = abs(A - 1/10)
D = abs(B - 1/10)

C
D

check(C < D)
check(C == 1 / 2^26 / 10)
check(D == 1 / 2^24 / 10)
check(C == 1/4 D)

"ok"
