"Atomic transitions 3"

"Verify dimensions"

joule = kilogram meter^2 / second^2
e = coulomb
rho = joule second / meter^3
epsilon0 = coulomb^2 / (joule meter)
hbar = joule second
m = kilogram
omega = 1/second
Iab = kilogram meter / second

check(e^2 / (epsilon0 m^2 hbar^2) Iab^2 rho / omega^2 == 1/second)

"ok"
