\section*{Addition of angular momentum 2}

In spherical coordinates
\begin{align*}
L_x&=i\hbar\left(\sin\phi\frac{\partial}{\partial\theta}
+\cot\theta\cos\phi\frac{\partial}{\partial\phi}\right)
\\
L_y&=i\hbar\left(-\cos\phi\frac{\partial}{\partial\theta}
+\cot\theta\sin\phi\frac{\partial}{\partial\phi}\right)
\\
L_z&=-i\hbar\frac{\partial}{\partial\phi}
\end{align*}

and
\begin{equation*}
L^2=L_x^2+L_y^2+L_z^2
=-\hbar^2\left(\frac{\partial^2}{\partial\theta^2}
+\frac{\cos\theta}{\sin\theta}\frac{\partial}{\partial\theta}
+\frac{1}{\sin^2\theta}\frac{\partial^2}{\partial\phi^2}\right)
\end{equation*}

Recall that for spherical harmonic $Y_{lm}(\theta,\phi)$
\begin{equation*}
L_zY_{lm}=m\hbar Y_{lm}
\end{equation*}

Let $\chi_+$ and $\chi_-$ be spin basis states such that
\begin{equation*}
\chi_+=\begin{pmatrix}1\\0\end{pmatrix},
\quad
\chi_-=\begin{pmatrix}0\\1\end{pmatrix}
\end{equation*}

Then for $\Psi=Y_{lm}\chi_+$ we have
\begin{align*}
J_z\Psi
&=L_z\Psi+S_z\Psi
\\
&=m\hbar\Psi+\tfrac{1}{2}\hbar\Psi
\\
&=\left(m+\tfrac{1}{2}\right)\hbar\Psi\tag{1}
\end{align*}

and for $\Psi=Y_{lm}\chi_-$
\begin{align*}
J_z\Psi
&=L_z\Psi+S_z\Psi
\\
&=m\hbar\Psi-\tfrac{1}{2}\hbar\Psi
\\
&=\left(m-\tfrac{1}{2}\right)\hbar\Psi\tag{2}
\end{align*}
