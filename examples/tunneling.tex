\section*{Tunneling}

Let $V(x)$ be the barrier potential
\begin{equation*}
V(x)=\begin{cases}
0 & x<-a
\\
V_0 & -a\le x\le a
\\
0 & x>a
\end{cases}
\end{equation*}

Find tunneling probability $T$ for a particle with mass $m$ and energy $E<V_0$.

\subsubsection*{\S1}

We have the following Schrodinger equations, one for each region of $V(x)$.
\begin{align*}
-\frac{\hbar^2}{2m}\frac{d^2}{dx^2}&\psi_1=E\psi_1 & x&\le-a
\\[1ex]
-\frac{\hbar^2}{2m}\frac{d^2}{dx^2}&\psi_2+V_0\psi_2=E\psi_2 & -a&\le x\le a
\\[1ex]
-\frac{\hbar^2}{2m}\frac{d^2}{dx^2}&\psi_3=E\psi_3 & x&\ge a
\end{align*}

The solutions are
\begin{align*}
\psi_1(x)&=A\exp(ikx)+B\exp(-ikx)
\\
\psi_2(x)&=C\exp(i\beta x)+D\exp(-i\beta x)
\\
\psi_3(x)&=F\exp(ikx)
\end{align*}

where
\begin{equation*}
k=\frac{\sqrt{2mE}}{\hbar}
\qquad
\beta=\frac{\sqrt{2m(E-V_0)}}{\hbar}
\end{equation*}

Boundary conditions at $x=-a$.
\begin{align*}
\psi_1(-a)&=\psi_2(-a)
\\
\psi_1'(-a)&=\psi_2'(-a)
\end{align*}

Boundary conditions at $x=a$.
\begin{align*}
\psi_2(a)&=\psi_3(a)
\\
\psi_2'(a)&=\psi_3'(a)
\end{align*}

From boundary conditions for $\psi_2$ and $\psi_3$ we have
\begin{equation*}
C\exp(i\beta a)+D\exp(-i\beta a)=F\exp(ikx)\tag{1}
\end{equation*}

and
\begin{equation*}
i\beta C\exp(i\beta a)-i\beta D\exp(-i\beta a)=ikF\exp(ikx)\tag{2}
\end{equation*}

Add $i\beta$ times (1) to (2) to obtain
\begin{equation*}
2i\beta C\exp(i\beta a)=(i\beta+ik)F\exp(ika)
\end{equation*}

Hence
\begin{equation*}
C=\frac{(\beta+k)F\exp(ika-i\beta a)}{2\beta}
\end{equation*}

Add minus $i\beta$ times (1) to (2) to obtain
\begin{equation*}
-2i\beta D\exp(-i\beta a)=(-i\beta+ik)F\exp(ika)
\end{equation*}

Hence
\begin{equation*}
D=\frac{(\beta-k)F\exp(ika+i\beta a)}{2\beta}
\end{equation*}

From boundary conditions for $\psi_1$ and $\psi_2$ we have
\begin{equation*}
A\exp(-ika)+B\exp(ika)=C\exp(-i\beta a)+D\exp(i\beta a)\tag{3}
\end{equation*}

and
\begin{equation*}
ikA\exp(-ika)-ikB\exp(ika)=i\beta C\exp(-i\beta a)-i\beta D\exp(i\beta a)\tag{4}
\end{equation*}

Add $ik$ times (3) to (4) to obtain
\begin{equation*}
2ikA\exp(-ika)=(ik+i\beta)C\exp(-i\beta a)+(ik-i\beta)D\exp(i\beta a)
\end{equation*}

Hence
\begin{equation*}
A=\frac{(k+\beta)C\exp(ika-i\beta a)}{2k}+\frac{(k-\beta)D\exp(ika+i\beta a)}{2k}
\end{equation*}

\iffalse

Add minus $ik$ times (3) to (4) to obtain
\begin{equation*}
-2ikB\exp(ika)=(-ik+i\beta)C\exp(-i\beta a)+(-ik-i\beta)D\exp(i\beta a)
\end{equation*}

Hence
\begin{equation*}
B=\frac{(k-\beta)C\exp(-ika-i\beta a)}{2k}+\frac{(k+\beta)D\exp(-ika+i\beta a)}{2k}
\end{equation*}

\fi

Substitute for $C$ and $D$.
\begin{equation*}
A=\frac{(k+\beta)^2F\exp[2ia(k-\beta)]}{4k\beta}
-\frac{(k-\beta)^2F\exp[2ia(k+\beta)]}{4k\beta}
\end{equation*}

Hence
\begin{equation*}
\frac{A}{F}=\frac{(k+\beta)^2\exp[2ia(k-\beta)]}{4k\beta}
-\frac{(k-\beta)^2\exp[2ia(k+\beta)]}{4k\beta}
\tag{5}
\end{equation*}

\subsubsection*{\S2}

Substituting for $k$ and $\beta$ we have for tunneling probability $T$
\begin{equation*}
T^{-1}=\frac{A}{F}\left(\frac{A}{F}\right)^*
=1+\frac{1}{8}\left(\frac{E}{V_0-E}+\frac{V_0}{E}+1\right)
\left[\cos\left(\frac{4a}{\hbar}\sqrt{2m(E-V_0)}\right)-1\right]
\tag{6}
\end{equation*}

Transform cos to sinh.
\begin{equation*}
T^{-1}=1+\frac{1}{4}\left(\frac{E}{V_0-E}+\frac{V_0}{E}+1\right)
\sinh^2\left(\frac{2ia}{\hbar}\sqrt{2m(E-V_0)}\right)
\tag{7}
\end{equation*}

Cancel the imaginary unit by swapping $E$ and $V_0$.
\begin{equation*}
T^{-1}=1+\frac{1}{4}\left(\frac{E}{V_0-E}+\frac{V_0}{E}+1\right)
\sinh^2\left(\frac{2a}{\hbar}\sqrt{2m(V_0-E)}\right)
\end{equation*}

Equivalently
\begin{equation*}
T^{-1}=1+\frac{V_0^2}{4E(V_0-E)}\sinh^2\left(\frac{2a}{\hbar}\sqrt{2m(V_0-E)}\right)
\tag{8}
\end{equation*}

Check dimensions.
Square root of kilogram joules is momentum.
\begin{equation*}
\frac{2a}{\hbar}\sqrt{2m(V_0-E)}=\frac{[\text{m}]}{[\text{J}\,\text{s}]}
\sqrt{[\text{kg}]\,[\text{J}]}=[1]
\tag{9}
\end{equation*}
