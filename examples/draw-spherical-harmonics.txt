# Draw |Y|^2

Y(l,m) = (-1)^m sqrt((2 l + 1) / (4 pi) (l - m)! / (l + m)!) P(l,m) exp(i m phi)

P(l,m,k) = test(m < 0, (-1)^m (l + m)! / (l - m)! P(l,-m),
           (sin(theta)/2)^m sum(k, 0, l - m,
           (-1)^k (l + m + k)! / (l - m - k)! / (m + k)! / k! *
           ((1 - cos(theta)) / 2)^k))

xrange = (-1,1) / 2
yrange = (-1,1) / 2

u = (sin(theta),cos(theta))

y = Y(2,2)
f = y conj(y) u
draw(f,theta)

y = Y(2,1)
f = y conj(y) u
draw(f,theta)

y = Y(2,0)
f = y conj(y) u
draw(f,theta)
