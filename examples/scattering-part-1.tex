\section*{Scattering part 1}

Let $\psi(r,\theta,\phi)$ be the Born approximation
\begin{equation*}
\psi(r,\theta,\phi)=\frac{e^{ikr}}{r}f(\theta,\phi)
\end{equation*}

where
\begin{equation*}
k=\frac{\sqrt{2mE}}{\hbar}
\end{equation*}

and $f(\theta,\phi)$ is the scattering amplitude
\begin{equation*}
f(\theta,\phi)=-\frac{m}{2\pi\hbar^2}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf x}{\hbar}\right)V(\mathbf x)\,d\mathbf x
\end{equation*}

Born approximation density is
\begin{equation*}
|\psi(r,\theta,\phi)|^2=\frac{|f(\theta,\phi)|^2}{r^2}
\end{equation*}

Differential cross sections are completely determined by the scattering amplitude.
\begin{equation*}
\frac{d\sigma}{d\Omega}=\left|\psi(r,\theta,\phi)\right|^2r^2=|f(\theta,\phi)|^2
\end{equation*}

For example, for Coulomb potentials we have
\begin{equation*}
f(\theta,\phi)=\frac{Z\alpha\hbar c}{2E(1-\cos\theta)}
\end{equation*}

Hence the Rutherford cross section
\begin{equation*}
\frac{d\sigma}{d\Omega}=\left[\frac{Z\alpha\hbar c}{2E(1-\cos\theta)}\right]^2
\end{equation*}

Verify dimensions of exponential.
\begin{equation*}
\frac{\sqrt{2mE}}{\hbar}\,r
=\frac{\sqrt{[\text{kg}]\,[\text J]}}{[\text J\,\text s]}\,[\text m]=[1]
\end{equation*}

Verify dimensions of cross section.
\begin{equation*}
\left(\frac{Z\alpha\hbar c}{2E}\right)^2
=\left(\frac{[\text J\,\text s]\,[\text m\,\text s^{-1}]}{[\text J]}\right)^2=[\text m^2]
\end{equation*}
