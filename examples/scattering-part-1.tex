\section*{Scattering part 1}

Let $\psi(r,\theta,\phi)$ be the Born approximation
\begin{equation*}
\psi(r,\theta,\phi)=\frac{e^{ikr}}{r}f(\theta,\phi)
\end{equation*}

where
\begin{equation*}
k=\frac{\sqrt{2mE}}{\hbar}
\end{equation*}

and
\begin{equation*}
f(\theta,\phi)=-\frac{m}{2\pi\hbar^2}
\int\exp\left(\frac{i\mathbf p\cdot\mathbf x}{\hbar}\right)V(\mathbf x)\,d\mathbf x
\end{equation*}

The density is
\begin{equation*}
|\psi|^2=\psi^*\psi=\frac{|f(\theta,\phi)|^2}{r^2}
\end{equation*}

Let $d\sigma$ be the differential cross section such that
\begin{equation*}
d\sigma=|\psi|^2\,dA=|\psi|^2r^2\,d\Omega=|f(\theta,\phi)|^2\,d\Omega
\end{equation*}

Hence the cross section is completely defined by $f(\theta,\phi)$ from the wave function.
\begin{equation*}
\frac{d\sigma}{d\Omega}=|f(\theta,\phi)|^2
\end{equation*}

For example, for Coulomb potentials we have the Born approximation
\begin{equation*}
\psi(r,\theta,\phi)=\frac{e^{ikr}}{r}\frac{Z\alpha\hbar c}{2E(1-\cos\theta)}
\end{equation*}

Hence the Rutherford cross section formula
\begin{equation*}
\frac{d\sigma}{d\Omega}=\left[\frac{Z\alpha\hbar c}{2E(1-\cos\theta)}\right]^2
\end{equation*}

Verify dimensions of exponential.
\begin{equation*}
\frac{\sqrt{2mE}}{\hbar}\,r
=\frac{\sqrt{[\text{kg}]\,[\text J]}}{[\text J\,\text s]}\,[\text m]=[1]
\end{equation*}

Verify dimensions of cross section.
\begin{equation*}
\left(\frac{Z\alpha\hbar c}{2E}\right)^2
=\left(\frac{[\text J\,\text s]\,[\text m\,\text s^{-1}]}{[\text J]}\right)^2=[\text m^2]
\end{equation*}
