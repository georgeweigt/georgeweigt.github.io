-- Feynman and Hibbs problem 4-12

xa = (xa1,xa2,xa3)
xb = (xb1,xb2,xb3)

X = xb - xa

"Verify equation (1)"

A = -i / (2 m hbar) T
B = i / hbar X[1]

check(-pi / A == -2 pi i m hbar / T)
check(-B^2 / (4 A) == i m X[1]^2 / (2 hbar T))

"ok"

"Verify equation (2)"

K = product(k,1,3,sqrt(-2 pi i m hbar / T) exp(i m X[k]^2 / (2 hbar T)))

K = K / (2 pi hbar)^3

check(K == (-i m / (2 pi hbar T))^(3/2) exp(i m dot(X,X) / (2 hbar T)))

"ok"
