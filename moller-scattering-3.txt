-- Draw probability density for high energy Moller scattering

-- eigenmath.org/moller-scattering-3.txt

pi = float(pi) -- use numerical value of pi
N = 6 -- 30 degree bins
a = pi / N

I(xi) = cos(a) + 8 cos(a)/sin(a)^2 - cos(xi) - 8 cos(xi)/sin(xi)^2

C = I(pi - a)

F(theta) = I(theta) / C

f = 1/C (3 + cos(theta)^2)^2 / sin(theta)^3

xrange = (a,pi - a)
yrange = (0,2)
draw(f,theta)

P = zero(N)

for(k,1,N - 2,
  theta1 = k a,
  theta2 = theta1 + a,
  P[k + 1] = F(theta2) - F(theta1)
)

P
