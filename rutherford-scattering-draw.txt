-- Draw cdf for Rutherford scattering

N = 10 -- number of bins

I = 1 / (cos(theta) - 1)
a = pi / N
C = eval(I,theta,pi) - eval(I,theta,a)
F = (I - eval(I,theta,a)) / C

"Cumulative distribution function"

xrange = (0,pi)
yrange = (0,1)
draw(F,theta)

"Bin probability (first bin not observed)"

P = zero(N)
for(k,2,N, P[k] = eval(F,theta,k pi/N) - eval(F,theta,(k-1) pi/N))
h(x) = test(x < 1, 0, x >= N + 1, 0, P[floor(x)])
xrange = (1,N+1)
yrange = (0,1)
draw(h,x)
