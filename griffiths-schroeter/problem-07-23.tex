\FBOX{
7.23
Use Equation 7.60 to estimate the internal field in hydrogen, and characterize
quantitatively a ``strong'' and ``weak'' Zeeman field.
}

Recall that
\begin{equation*}
|\mathbf L|=\sqrt{l(l+1)}\hbar,\quad l=0,1,2,\ldots n-1
\end{equation*}

Let $n=2$, $l=1$, and
\begin{equation*}
r=\frac{3n^2-l(l+1)}{2}a_0=5a_0
\end{equation*}

Then by
\begin{equation*}
\mathbf B=\frac{1}{4\pi\epsilon_0}\frac{e}{mc^2r^3}\mathbf L
\tag{7.60}
\end{equation*}

we have
\begin{equation*}
|\mathbf B|\approx0.14\,\text{tesla}
\end{equation*}

Note that for $n=1$ we have $l=0$ and $|\mathbf B|=0$.
