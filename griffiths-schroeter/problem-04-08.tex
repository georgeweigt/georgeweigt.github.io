\FBOX{
4.8
Starting from the Rodrigues formula, derive the orthonormality condition for
Legendre polynomials:
\begin{equation*}
\int_{-1}^1P_\ell(x)P_{\ell'}(x)\,dx=\left(\frac{2}{2\ell+1}\right)
\delta_{\ell\ell'}.\tag{4.34}
\end{equation*}

{\it Hint:} Use integration by parts.
}

This is the Rodrigues formula.
\begin{equation*}
P_\ell(x)=\frac{1}{2^\ell\ell!}\left(\frac{d}{dx}\right)^\ell
\left(x^2-1\right)^\ell
\end{equation*}

Binomial formula
\begin{equation*}
\left(x^2-1\right)^\ell=\sum_{k=0}^\ell\binom{\ell}{k}x^{2(\ell-k)}(-1)^k
\end{equation*}

Hence
\begin{equation*}
\left(\frac{d}{dx}\right)^\ell\left(x^2-1\right)^\ell
=\sum_{k=0}^\ell c_kx^{2(\ell-k)-\ell}
\end{equation*}
