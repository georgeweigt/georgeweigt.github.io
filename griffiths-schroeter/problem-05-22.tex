\FBOX{
5.22
Helium-3 is a fermion with spin 1/2 (unlike the more common isotope
helium-4 which is a boson).
At low temperatures ($T\ll T_F$), helium-3 can be treated as a Fermi gas
(Section 5.3.1).
Given a density of $82\,\text{kg/m}^3$, calculate $T_F$
(Problem 5.21(c)) for helium-3.
}

\begin{equation*}
\rho=\frac{82\,\text{kg/m}^3}{3m_p}
=1.64\times10^{28}\,\text{m}^{-3}
\end{equation*}

For
\begin{align*}
\hbar&=1.05\times10^{-34}\,\text{J s}
\\
m&=9.11\times10^{-31}\,\text{kg}
\\
k_B&=1.38\times10^{-23}\,\text{J/K}
\end{align*}

we have
\begin{equation*}
E_F=\frac{\hbar^2}{2m}\left(3\pi^2\rho\right)^{2/3}=7.80\times10^{-20}\,\text{J}
\end{equation*}

and
\begin{equation*}
T_F=\frac{E_F}{k_B}=5653\,\text{K}
\end{equation*}
